import os
from docx import Document

# 1. 경로 설정 (사용자님의 환경에 맞춤)
source_dir = r'C:\Users\Downloads\자기소개서'  # 원본 폴더
target_dir = r'C:\Users\Downloads\자기소개서_텍스트' # 변환될 텍스트 저장 폴더

# 저장할 폴더가 없으면 새로 만듭니다.
if not os.path.exists(target_dir):
    os.makedirs(target_dir)

print("텍스트 추출을 시작합니다...")

# 2. 하위 폴더까지 모두 뒤지는 로직 (os.walk)
for root, dirs, files in os.walk(source_dir):
    for filename in files:
        if filename.endswith(".docx") and not filename.startswith("~$"):
            try:
                # 워드 파일 열기
                doc_path = os.path.join(root, filename)
                doc = Document(doc_path)
                
                # 텍스트 추출
                full_text = [para.text for para in doc.paragraphs]
                
                # 결과 파일명 설정 (중복 방지를 위해 연도 폴더명을 앞에 붙여주면 더 좋습니다)
                # 예: 2024_삼성전자.txt
                rel_path = os.path.relpath(root, source_dir)
                folder_prefix = rel_path.replace(os.sep, "_")
                
                if folder_prefix == ".": # 루트 폴더에 있는 경우
                    new_filename = filename.replace(".docx", ".txt")
                else:
                    new_filename = f"{folder_prefix}_{filename.replace('.docx', '.txt')}"
                
                # 텍스트 파일로 저장
                with open(os.path.join(target_dir, new_filename), "w", encoding="utf-8") as f:
                    f.write("\n".join(full_text))
                
                print(f"성공: {filename} -> {new_filename}")
            
            except Exception as e:
                print(f"오류 발생 ({filename}): {e}")

print("-" * 30)
print(f"모든 작업이 완료되었습니다! 파일은 다음 폴더에 있습니다: {target_dir}")
